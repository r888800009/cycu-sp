	PROGRAM A1;
	VARIABLE INTEGER:X,Y,i1,i2,i3;
	LABEL L1 ;
	IF  X EQ Y  THEN  X=10;
	LABEL L2 ;
	IF  X NE 100  THEN  X=10+20;
	LABEL L3 ;
	IF  100 GT Y  THEN  X=10*20;
	LABEL L4 ;
	IF  X GE Y  THEN  X=10↑20;
	LABEL L5 ;
	IF  X LT Y  THEN  X=10*20-30+40/50;
	LABEL L6 ;
	IF  X LE Y  THEN  X=10+20*30/40-50;
	LABEL L7 ;
	IF  X EQ Y  THEN  X=(i1-i2)↑2+i3;
	LABEL L8 ;
	IF  X NE Y  THEN  ;
	LABEL LL ;
	ENP;
